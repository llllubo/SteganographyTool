 41    @staticmethod
 42    def __opcode_correction(instr: MyInstruction, opcode: int) -> str:
 43        """
 44        Correction of read OPCODE to the desired format acceptable for
 45        decode. There is one special case where OPCODE has to be
 46        modified, additionally. It's when 1 byte register operand is
 47        used.
 48        
 49        This correction is applied for classes 'SUB/XOR' and
 50        'TEST/AND/OR'.
 51        """
 52        if OpCodeInfo(instr.instruction.code).operand_size == 0:
 53            # Correction of 1 byte operand OPCODE.
 54            # Endianness does not matter as OPCODE is always 1 byte long.
 55            opcode += 1
 56            return f"{opcode:#04x}"
 57        else:
 58            return f"{opcode:#04x}"

 61    @staticmethod
 62    def __get_order_type(instr: MyInstruction, next_mov: MyInstruction) -> str:
 63        """
 64        Determine occuring order. It's designed for classes
 65        'MOV Scheduling' and 'Swap base-index registers 32-bit' as
 66        only these use scheduling.
 67        
 68        Only in case of class 'MOV Scheduling' is 2nd parameter given,
 69        otherwise it's set to None. This parameter is 1st occured MOV.
 70        """
 71        
 72        if next_mov is not None:
 73            if instr.eq_class.class_name == "MOV Scheduling" or \
 74                next_mov.eq_class.class_name == "MOV Scheduling" or \
 75                instr.mov_scheduling_flag or next_mov.mov_scheduling_flag:
 76                mov0 = f"{instr.instruction}"
 77                mov1 = f"{next_mov.instruction}"
 78                
 79                if common.lexicographic_mov_sort(mov0) < \
 80                    common.lexicographic_mov_sort(mov1):
 81                    return "Ascending"
 82                else:
 83                    return "Descending"
 84            
 85        elif instr.eq_class.class_name == "Swap base-index registers 32-bit":
 86            if instr.instruction.memory_base < instr.instruction.memory_index:
 87                return "Ascending"
 88            else:
 89                return "Descending"

 92    @staticmethod
 93    def __decode_mov(order: str) -> bitarray:
 94        """
 95        Decode given order. For 'MOV Scheduling' can occur situation
 96        when both MOVs will have set only `mov_scheduling_flag`.
 97        Therefore can not be used classic `__decode()` function and
 98        this special one was created.
 99        """
100        for eq_class in EqClassesProcessor.all_eq_classes:
101            if eq_class.class_name == "MOV Scheduling":
102                for idx, mem in enumerate(eq_class.members):
103                    if order == mem:
104                        return eq_class.encoded_idxs[idx]

109        """
110        Decode extracted bits to message bits. This is determined by
111        encoding defined in configuration file.
112        
113        `extracted_mem` can be of more data types. It depends on
114        equivalent class.

116        for idx, mem in enumerate(eq_class.members):
117            if extracted_mem == mem:
118                return eq_class.encoded_idxs[idx]

121    @classmethod
122    def extract(cls,
123                fexe: str,
124                potential_my_instrs: list,
125                verbose: bool) -> bitarray:
126        """
127        Extract secret message.
128        """
129        try:
130            fd = open(fexe, "rb")
131        except IOError:
132            print(f"ERROR! Can not open stego-file for extracting: {fexe}",
133                  file=sys.stderr)
134            sys.exit(105)
135        
136        # Prepared bitarray to store extracted bits here. At the end it
137        # will be returned by this function.
138        bits_mess = bitarray(endian="little")
139        # Extract limit defines how many bits are going to be extracted.
140        # This limit changes during the extraction as first it's set to
141        # amount of bits which defines length of further data. After
142        # this limit is reached, second is set that determines length of
143        # real data. If limit will be reached, fulfilled bitarray is
144        # returned. If the end of potential instructions will come first
145        # it's known that it can not be extracted all data, because of
146        # insufficient capacity of executable. In that case all
147        # collected bits are returned and flag is set.
148        extract_limit = common.SIZE_OF_DATA_LEN * 8
149        # This flag determines phase of extraction.
150        # Two possible values:
151        #   * False -> extracting is first bytes storing data length.
152        #   * True  -> extracting are real data (payload).
153        data_extraction_flag = False
154        # Skip flag defines how many next instructions should be
155        # skipped as they were already used by first instruction
156        # from their group (3 and 2 Bytes Long NOP classes).
157        skip = 0        
158        
159        # Determines if MOVs were already scheduled. It's False at the
160        # beginning, but first MOV set it to True after they are
161        # scheduled.
162        movs_scheduled = False
163            
164        for instr_idx, my_instr in enumerate(potential_my_instrs):
165            
166            if skip:
167                # Skip current NOP instruction.
168                skip -= 1
169                continue
170            
171            # For speed performance.
172            eq_class = my_instr.eq_class
173            instr = my_instr.instruction            
174            
175            # Instructions that don't have encoding LEGACY are skipped.
176            # Legacy encoding for opcodes of instructions is mainly used
177            # in each PE and ELF (32- and 64-bit) executables and others
178            # encodings are very rare.
179            if eq_class is not None and \
180                instr.encoding == EncodingKind.LEGACY:
181                
182                # Encoding is lexicographic order of used instructions
183                # strings and it's determined by configuration file.
184                if eq_class.class_name == "MOV Scheduling" or \
185                    my_instr.mov_scheduling_flag and not movs_scheduled:
186
187                    ## Second MOV is current, scheduling can be decoded.
188
189                    # Set flag with first MOV - This is only tag for
190                    # knowledge that first MOV was tried to be scheduled.
191                    movs_scheduled = True
192
193                    # Get type of order in which are both MOVs present.
194                    order = cls.__get_order_type(my_instr,
195                                                 potential_my_instrs[instr_idx +1])
196                    # Decode found out order. Here can occur situation
197                    # when both MOVs will have set only
198                    # 'mov_scheduling_flag'. Therefore can not be used
199                    # '__decode()' function here and special one was
200                    # created.
201                    extracted_bits = cls.__decode_mov(order)
202                    
203                    # Collect decoded bits and create message.
204                    bits_mess.extend(extracted_bits)
205                else:
206                    # Unset flag with second MOV.
207                    movs_scheduled = False
208                    
209                
210                if re.match(r"^(?:MOV|ADD|SUB|AND|OR|XOR|CMP|ADC|SBB)$",
211                          eq_class.class_name) or \
212                    re.match(r"^(?:ADD|SUB|AND|OR|XOR|CMP|ADC|SBB) 32-bit$",
213                          eq_class.class_name):
214                    # Secret bit is stored at the place of Direction bit
215                    # inside OPCODE.
216                    # MOV instruction from this class can also be
217                    # scheduled - NOT IMPLEMENTED.
218                    
219                    # Read instruction bytes from file to be able to
220                    # analyze it.
221                    fd.seek(my_instr.foffset)
222                    b_instr_fromf = fd.read(len(instr))
223                    
224                    # Convert read instruction from bytes to bits.
225                    bits_instr = bitarray()
226                    bits_instr.frombytes(b_instr_fromf)
227                    
228                    # Get and find an opcode of instruction.
229                    instr_opcode = OpCodeInfo(instr.code).op_code
230                    opcode_idx = common.get_opcode_idx(b_instr_fromf,
231                                                       instr_opcode)
232                    
233                    dir_bit_offset = (opcode_idx * 8) + 6
234                    # Decode read Direction bit.
235                    extracted_bits = cls.__decode(eq_class,
236                                bits_instr[dir_bit_offset:dir_bit_offset + 1])
237                    
238                    # Collect decoded bits and create message.
239                    bits_mess.extend(extracted_bits)
240
241
242                # Encoding is lexicographic order of used registers name
243                # and it's determined by configuration file.
244                if eq_class.class_name == "Swap base-index registers 32-bit":
245                    # Instruction form changes (SIB.base <=> SIB.index),
246                    # therefore, also, operands must be changed. If REX
247                    # prefix is present, proper bits of prefix are
248                    # exchanged, as well.
249                    # MOV instruction from this class can also be
250                    # scheduled - NOT IMPLEMENTED.
251
252                    # Get type of order in which are memory registers
253                    # present.
254                    order = cls.__get_order_type(my_instr, None)
255                    
256                    # Decode read Direction bit.
257                    extracted_bits = cls.__decode(eq_class, order)
258
259                    # Collect decoded bits and create message.
260                    bits_mess.extend(extracted_bits)
261                
262                
263                # Classes can be together as their usage is based on
264                # exactly same principle.
265                elif eq_class.class_name == "2 Bytes Long NOP" or \
266                    eq_class.class_name == "3 Bytes Long NOP":
267
268                    # Set skip flag for next instructions skipping.
269                    skip = common.set_skip_flag(my_instr,
270                                                potential_my_instrs[instr_idx + 1])
271                    # Read instruction bytes from file to be able to
272                    # analyze it.
273                    fd.seek(my_instr.foffset)
274                    if eq_class.class_name == "2 Bytes Long NOP":
275                        b_instr_fromf = fd.read(2)
276                    elif eq_class.class_name == "3 Bytes Long NOP":
277                        b_instr_fromf = fd.read(3)
278                    
279                    # Convert bytes to hex string.
280                    hex_instr = "0x" + b_instr_fromf.hex()
281                    
282                    # Decode read instruction.
283                    extracted_bits = bitarray(endian="little")
284                    extracted_bits = cls.__decode(eq_class, hex_instr)
285                    
286                    # Collect decoded bits and create message.
287                    bits_mess.extend(extracted_bits)
288
289
290                # Class does not encodes class members, as it does not
291                # have any. In this case, bits from message are simply
292                # embedded to the last useable instruction bytes.
293                elif eq_class.class_name == ">3 Bytes Long NOP":
294
295                    # Get number of last instruction bits which contain
296                    # message bits.
297                    fd.seek(my_instr.foffset)
298                    bits_cnt = \
299                        common.count_useable_bits_from_nop(instr,
300                                                           fd.read(len(instr)))
301                    # There is 100% chance that it will be multiple of 8.
302                    b_cnt = bits_cnt // 8
303                    
304                    # Extract from executable.
305                    pos = my_instr.foffset + (len(instr) - b_cnt)
306                    fd.seek(pos)
307                    b_extracted = fd.read(b_cnt)
308                    
309                    # Convert extracted bytes to bits.
310                    extracted_bits = bitarray(endian="little")
311                    extracted_bits.frombytes(b_extracted)
312
313                    # Collect decoded bits and create message.
314                    bits_mess.extend(extracted_bits)
315                    
316                
317                # These two classes can be merged as they modify only
318                # Reg/Opcode field inside ModR/M byte.
319                elif eq_class.class_name == "SHL/SAL" or \
320                    eq_class.class_name == "TEST non-accumulator register":
321                    # Secret bit is defined by Reg/Opcode bits of ModR/M
322                    # byte inside instruction.
323
324                    # Read instruction bytes from file to be able to
325                    # analyze it.
326                    fd.seek(my_instr.foffset)
327                    b_instr_fromf = fd.read(len(instr))
328
329                    # Convert read instruction from bytes to bits.
330                    bits_instr = bitarray()
331                    bits_instr.frombytes(b_instr_fromf)
332
333                    # Get and find an opcode of instruction.
334                    instr_opcode = OpCodeInfo(instr.code).op_code
335                    opcode_idx = common.get_opcode_idx(b_instr_fromf,
336                                                       instr_opcode)
337                    
338                    # Locate Reg/Opcode field inside ModR/M byte.
339                    reg_field_offset = (opcode_idx + 1) * 8 + 2
340                    
341                    # Decode read Direction bit.
342                    extracted_bits = cls.__decode(eq_class,
343                            bits_instr[reg_field_offset:reg_field_offset + 3])
344
345                    # Collect decoded bits and create message.
346                    bits_mess.extend(extracted_bits)
347                
348                
349                elif eq_class.class_name == "SUB/XOR" or \
350                    eq_class.class_name == "TEST/AND/OR":
351
352                    # Read instruction bytes from file to be able to
353                    # analyze it.
354                    fd.seek(my_instr.foffset)
355                    b_instr_fromf = fd.read(len(instr))
356
357                    # Convert read instruction from bytes to bits.
358                    bits_instr = bitarray()
359                    bits_instr.frombytes(b_instr_fromf)
360                    
361                     # Get and find an opcode of instruction.
362                    instr_opcode = OpCodeInfo(instr.code).op_code
363
364                    # Correctness of read OPCODE.
365                    opcode = cls.__opcode_correction(my_instr, instr_opcode)
366
367                    # Decode read Direction bit.
368                    extracted_bits = cls.__decode(eq_class, opcode)
369
370                    # Collect decoded bits and create message.
371                    bits_mess.extend(extracted_bits)
372                    
373                
374                elif eq_class.class_name == "ADD negated" or \
375                    eq_class.class_name == "SUB negated":
376                    pass
377
378                
379            else:
380                print(f"CAN NOT BE USED - NOT LEGACY INSTRUCTION.", file=sys.stderr)
381            
382            
383            if len(bits_mess) >= extract_limit:
384                
385                if not data_extraction_flag:
386                    # If bits defined data length are available, their
387                    # length is decoded and set as new extract limit.
388                    b_xored_len = bits_mess[:extract_limit].tobytes()
389                    
390                    # UnXOR extracted length of data with password.
391                    data_len = cls.__unxor_data_len(b_xored_len)
392                    
393                    # Remove bits specifying data length from array.
394                    del bits_mess[:extract_limit]
395                    
396                    # Set new extraction limit (file extension + raw
397                    # data).
398                    extract_limit = data_len * 8 + common.SIZE_OF_FEXT * 8
399                    
400                    # Set next extracting phase.
401                    data_extraction_flag = True
402                else:
403                    # Full requested message was extracted.
404                    if verbose:
405                        print("All required data was extracted.")
406                    break
407    
408        fd.close()
409    
410        # Check if all expected data were extracted.
411        if len(bits_mess) < extract_limit:
412            # Not all requested data could be extracted.
413            # This should not happen and if do, it will throw error
414            # when Fernet checks integrity of data,
415            pass
416        
417        # Correctness of extracted data. Bits extracted in addition,
418        # was extracted with last required bits and must be truncated.
419        del bits_mess[extract_limit:]
420        
421        return bits_mess

424    @classmethod
425    def __unxor_data_len(cls, xored_len: bytes) -> int:
426        """
427        Function returns XORed data length as integer and also it
428        requires password from user.
429        """
430        cls.__b_key, cls.__b_passwd = common.gen_key_from_passwd()
431        
432        # Prepare password length ONLY for XOR operation.
433        if len(cls.__b_passwd) <= common.SIZE_OF_DATA_LEN:
434            b_passwd = cls.__b_passwd
435            b_passwd += bytes(common.SIZE_OF_DATA_LEN - len(cls.__b_passwd))
436        else:
437            b_passwd = cls.__b_passwd[:common.SIZE_OF_DATA_LEN]
438
439        # Data length bytes are going to be unXORed with password
440        # of same size. If given password is longer, it's truncated.
441        # This is security reason as if whole password bytes were
442        # XORed with null padding of data length bytes, password
443        # characters would map to the XORed data length and could be
444        # readable. Reversed cycle is used to avoid problem when
445        # null byte is in the middle of data.
446        null_padding = 0
447        
448        for b in reversed(xored_len):
449            if b != 0:
450                break
451            null_padding += 1
452        
453        i = common.SIZE_OF_DATA_LEN - null_padding
454        
455        b_unxored_len = bytes([a ^ b for a, b in zip(xored_len, b_passwd[:i])])
456        
457        # Add null bytes after XOR.
458        b_unxored_len += bytes(common.SIZE_OF_DATA_LEN - len(b_unxored_len))
459        
460        return int.from_bytes(b_unxored_len, byteorder="little")

463    @classmethod
464    def unxor_fext(cls, xored_fext: bytes) -> bytes:
465        """
466        Function returns unXORed file extension in bytes of proposed
467        length.
468        """
469        
470        # Check if there is any file extension, if not, return given
471        # bytes.
472        if xored_fext != bytes(common.SIZE_OF_FEXT):
473            
474            # Prepare password length ONLY for XOR operation.
475            if len(cls.__b_passwd) <= common.SIZE_OF_FEXT:
476                b_passwd = cls.__b_passwd
477                b_passwd += bytes(common.SIZE_OF_FEXT - len(cls.__b_passwd))
478            else:
479                b_passwd = cls.__b_passwd[:common.SIZE_OF_FEXT]
480            
481            # File extension bytes are going to be unXORed with password
482            # of same size. If given password is longer, it's truncated.
483            # This is security reason as if whole password bytes were
484            # XORed with null padding of data length bytes, password
485            # characters would map to the XORed data length and could be
486            # readable. Reversed cycle is used to avoid problem when
487            # null byte is in the middle of data.
488            null_padding = 0
489            
490            for b in reversed(xored_fext):
491                if b != 0:
492                    break
493                null_padding += 1
494            
495            i = common.SIZE_OF_FEXT - null_padding
496            
497            b_unxored_fext = bytes([a ^ b for a, b in zip(xored_fext, b_passwd[:i])])
498            
499            # Add null bytes after XOR.
500            b_unxored_fext += bytes(common.SIZE_OF_FEXT - len(b_unxored_fext))
501
502            return b_unxored_fext
503        
504        return xored_fext

507    @classmethod
508    def decrypt_data(cls, data: bytes) -> bytes:
509        """
510        Decrypt given data in bytes.
511        """
512        cipher = Fernet(cls.__b_key)
513        try:
514            b_decrypted = cipher.decrypt(data)
515        except InvalidToken:
516            print("ERROR! Wrong password for extracting data.", file=sys.stderr)
517            sys.exit(105)
518
519        return b_decrypted

553    @classmethod
554    def make_output(cls, data: bytes, fext: bytes) -> None:
555        """
556        Ensure that output directory will be present.
557        """
558        try:
559            os.mkdir("./extracted")
560        except FileExistsError:
561            # Existing directory will be used.
562            pass
563        except OSError:
564            print("ERROR! Creation of output directory failed.", file=sys.stderr)
565            sys.exit(105)
566        
567        # Get rid of null bytes in file extension, if present.
568        ext = ""
569        for b in fext:
570            if b != 0:
571                ext += chr(b)
572        
573        # Add dot before extension, if present.
574        if ext != "":
575            ext = "." + ext
576            
577        fname = cls.__create_fname(ext)
578        
579        # Write extracted content to the file.
580        try:
581            fd = open(f"./extracted/{fname}", "wb")
582        except IOError:
583            print(f"ERROR! Can not access an output file: {fname}", file=sys.stderr)
584            sys.exit(105)
585        else:
586            fd.write(data)
587            fd.close()

425    @classmethod
426    def __unxor_data_len(cls, xored_len: bytes) -> int:
427        """
428        Function returns XORed data length as integer and also it
429        requires password from user.
430        """
431        cls.__b_key, cls.__b_passwd = common.gen_key_from_passwd()
432        
433        # Prepare password length ONLY for XOR operation.
434        if len(cls.__b_passwd) <= common.SIZE_OF_DATA_LEN:
435            b_passwd = cls.__b_passwd
436            b_passwd += bytes(common.SIZE_OF_DATA_LEN - len(cls.__b_passwd))
437        else:
438            b_passwd = cls.__b_passwd[:common.SIZE_OF_DATA_LEN]
439
440        # Data length bytes are going to be unXORed with password
441        # of same size. If given password is longer, it's truncated.
442        # This is security reason as if whole password bytes were
443        # XORed with null padding of data length bytes, password
444        # characters would map to the XORed data length and could be
445        # readable. Reversed cycle is used to avoid problem when
446        # null byte is in the middle of data.
447        null_padding = 0
448        
449        for b in reversed(xored_len):
450            if b != 0:
451                break
452            null_padding += 1
453        
454        i = common.SIZE_OF_DATA_LEN - null_padding
455        
456        b_unxored_len = bytes([a ^ b for a, b in zip(xored_len, b_passwd[:i])])
457        
458        # Add null bytes after XOR.
459        b_unxored_len += bytes(common.SIZE_OF_DATA_LEN - len(b_unxored_len))
460        
461        return int.from_bytes(b_unxored_len, byteorder="little")

467        Function returns unXORed file extension in bytes of proposed
468        length.
469        """
470        
471        # Check if there is any file extension, if not, return given
472        # bytes.
473        if xored_fext != bytes(common.SIZE_OF_FEXT):
474            
475            # Prepare password length ONLY for XOR operation.
476            if len(cls.__b_passwd) <= common.SIZE_OF_FEXT:
477                b_passwd = cls.__b_passwd
478                b_passwd += bytes(common.SIZE_OF_FEXT - len(cls.__b_passwd))
479            else:
480                b_passwd = cls.__b_passwd[:common.SIZE_OF_FEXT]
481            
482            # File extension bytes are going to be unXORed with password
483            # of same size. If given password is longer, it's truncated.
484            # This is security reason as if whole password bytes were
485            # XORed with null padding of data length bytes, password
486            # characters would map to the XORed data length and could be
487            # readable. Reversed cycle is used to avoid problem when
488            # null byte is in the middle of data.
489            null_padding = 0
490            
491            for b in reversed(xored_fext):
492                if b != 0:
493                    break
494                null_padding += 1
495            
496            i = common.SIZE_OF_FEXT - null_padding
497            
498            b_unxored_fext = bytes([a ^ b for a, b in zip(xored_fext, b_passwd[:i])])
499            
500            # Add null bytes after XOR.
501            b_unxored_fext += bytes(common.SIZE_OF_FEXT - len(b_unxored_fext))
502
503            return b_unxored_fext
504        
505        return xored_fext
