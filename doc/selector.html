 100    @staticmethod
 101    def __liveness_flags_detection(all_instrs: list,
 102                                   my_instr: MyInstruction,
 103                                   rflags_to_check: int,
 104                                   force_flag: bool) -> bool:
 105        """
 106        Liveness detection of modified flags by replacement instruction.
 107        Check given flags if they are live until any instruction which
 108        modifies it come. Execution flow is respected while checking.
 109        Flag is live if instruction which reads it come earlier than
 110        instruction which modofies it. Basic checking is done until
 111        any jump is present. If 'force' flag was given by user,
 112        unconditional jumps are traced and this can achieve bigger
 113        capacity. The end of any function stops checking with positive
 114        result.
 115        """
 116        if rflags_to_check == RflagsBits.NONE:
 117            return True
 118        
 119        all_rflags = (RflagsBits.OF, RflagsBits.SF, RflagsBits.ZF,
 120                      RflagsBits.AF, RflagsBits.CF, RflagsBits.PF,
 121                      RflagsBits.DF, RflagsBits.IF, RflagsBits.AC,
 122                      RflagsBits.UIF)
 123        
 124        # Transformation of given flags all in one integer to the list
 125        # of proper flags values.
 126        check_rflags = [rflag
 127                        for rflag in all_rflags
 128                            if (rflag & rflags_to_check) != 0]
 129        
 130        # List for marking whether the flag has already been checked
 131        # (modified by another instruction).
 132        safe_rflags = [False] * len(check_rflags)
 133
 134        # In 'my_instr' is always present next instruction from
 135        # previous one.
 136        my_instr = all_instrs[my_instr.ioffset + 1]
 137        
 138        while True:
 139            
 140            rflags_read = my_instr.instruction.rflags_read
 141            rflags_modified = my_instr.instruction.rflags_modified
 142
 143            for idx, rflag in enumerate(check_rflags):
 144                # Liveness detection of checked flags.
 145                
 146                if (rflag & rflags_read) != 0:
 147                    # Instruction tests checked flag.
 148                    return False
 149                
 150                if (rflag & rflags_modified) != 0:
 151                    # Instruction modifies checked flag.
 152                    safe_rflags[idx] = True
 153                    
 154                    # End immediately when all flags are safe.
 155                    if False not in safe_rflags:
 156                        return True
 157            
 158            # Only because of clearer conditions.
 159            instr = my_instr.instruction
 160            
 161            if instr.flow_control == FlowControl.NEXT:
 162                # Execution flow follow next instruction.
 163                my_instr = all_instrs[my_instr.ioffset + 1]
 164            
 165            elif instr.flow_control == FlowControl.RETURN:
 166                # The end of a function.
 167                break
 168            
 169            # CALL|JMP near/far
 170            # ONLY if user specify by argument (very time-consuming).
 171            elif force_flag and \
 172                (
 173                    instr.flow_control == FlowControl.CALL or \
 174                    instr.flow_control == FlowControl.UNCONDITIONAL_BRANCH
 175                ):
 176                   
 177                if instr.op_count == 1:
 178                    if instr.op0_kind == OpKind.NEAR_BRANCH16 or \
 179                        instr.op0_kind == OpKind.NEAR_BRANCH32 or \
 180                        instr.op0_kind == OpKind.NEAR_BRANCH64:
 181
 182                        get_target = instr.near_branch_target
 183                    
 184                    elif instr.op0_kind == OpKind.FAR_BRANCH16 or \
 185                        instr.op0_kind == OpKind.FAR_BRANCH32:
 186                        
 187                        get_target = instr.far_branch_selector
 188                        
 189                    else:
 190                        return False
 191                
 192                else:
 193                    return False
 194                    
 195                found_instr = [my_instr 
 196                                for my_instr in all_instrs 
 197                                    if my_instr.instruction.ip == get_target]
 198                
 199                if found_instr:
 200                    my_instr = found_instr[0]
 201                else:
 202                    return False
 203            
 204            else:   
 205                return False
 206
 207        return True

 210    @staticmethod
 211    def __check_regs_dependency(bitness: int,
 212                                reg0: Register_,
 213                                reg1: Register_) -> bool:
 214        """
 215        Check if given registers are influenced by each other (e.g. EAX
 216        & RAX).
 217        """
 218        info0 = RegisterInfo(reg0)
 219        info1 = RegisterInfo(reg1)
 220        if bitness == 32:
 221            full_reg0 = info0.full_register32
 222            full_reg1 = info1.full_register32
 223        else:
 224            full_reg0 = info0.full_register
 225            full_reg1 = info1.full_register
 226            
 227        if full_reg0 == full_reg1:
 228            return True
 229        
 230        return False

 253    @classmethod
 254    def __check_independency(cls,
 255                             prev_mov: Instruction,
 256                             curr_mov: Instruction,
 257                             bitness: int) -> bool:
 258        """
 259        Two registers are dependent on each other if their order is
 260        irreplaceable.
 261        
 262        If write to any register is performed, it can not be used
 263        anymore (bith sides). If register occurs on left side as
 264        memory operand, it's reading from it and can be used next time
 265        (MOV rax, rax; MOV rax, [rax]).
 266        
 267        From specific register or memory location can be read more
 268        times. If it's writing to any memory location, this location
 269        can not be used anymore.
 270        """
 271        
 272        # Can not be exactly two same MOV instructions.
 273        if prev_mov.eq_all_bits(curr_mov):
 274            return False
 275        # In case when they are not equal in bits, because of they
 276        # differ only in Direction Bit as MOV can -- in that case they
 277        # actually are equal and can not be ordered lexicographically.
 278        elif f"{prev_mov:ix}" == f"{curr_mov:ix}":
 279            return False
 280        
 281        # If some mov write to the REG, this REG can not be used anymore
 282        # (except within same instruction).
 283        if prev_mov.op0_kind == OpKind.REGISTER:
 284            
 285            if curr_mov.op0_kind == OpKind.REGISTER and \
 286                cls.__check_regs_dependency(bitness, curr_mov.op0_register, prev_mov.op0_register):
 287                return False
 288            if curr_mov.op1_kind == OpKind.REGISTER and \
 289                cls.__check_regs_dependency(bitness, curr_mov.op1_register, prev_mov.op0_register):
 290                return False
 291            if (
 292                    curr_mov.op0_kind == OpKind.MEMORY or \
 293                    curr_mov.op1_kind == OpKind.MEMORY
 294                ) and \
 295                cls.__mem_op_use_reg(bitness, curr_mov, prev_mov.op0_register):
 296                return False
 297        
 298        if curr_mov.op0_kind == OpKind.REGISTER:
 299            
 300            if prev_mov.op1_kind == OpKind.REGISTER and \
 301                cls.__check_regs_dependency(bitness, prev_mov.op1_register, curr_mov.op0_register):
 302                    return False
 303            if (
 304                    prev_mov.op0_kind == OpKind.MEMORY or \
 305                    prev_mov.op1_kind == OpKind.MEMORY
 306                ) and \
 307                cls.__mem_op_use_reg(bitness, prev_mov, curr_mov.op0_register):
 308                return False
 309            
 310        # Same MEMORY locations on the 'cross'.
 311        if prev_mov.op0_kind == OpKind.MEMORY:
 312            if prev_mov.op0_kind == curr_mov.op1_kind and \
 313                cls.__check_regs_dependency(bitness, prev_mov.memory_base, curr_mov.memory_base) and \
 314                cls.__check_regs_dependency(bitness, prev_mov.memory_index, curr_mov.memory_index) and \
 315                prev_mov.memory_index_scale == curr_mov.memory_index_scale and \
 316                prev_mov.memory_displacement == curr_mov.memory_displacement:
 317                return False
 318        
 319        # Same MEMORY locations on the 'cross'.
 320        if prev_mov.op1_kind == OpKind.MEMORY:
 321            if prev_mov.op1_kind == curr_mov.op0_kind and \
 322                cls.__check_regs_dependency(bitness, prev_mov.memory_base, curr_mov.memory_base) and \
 323                cls.__check_regs_dependency(bitness, prev_mov.memory_index, curr_mov.memory_index) and \
 324                prev_mov.memory_index_scale == curr_mov.memory_index_scale and \
 325                prev_mov.memory_displacement == curr_mov.memory_displacement:
 326                return False
 327            
 328        return True

 343    def select(cls,
 344               all_my_instrs: list,
 345               method: str,
 346               force_flag: bool,
 347               fexe: str,
 348               analyzer: Analyzer) -> list:
 349        """
 350        Select instructions according to the used equivalent classes.
 351        """
 352        selected_my_instrs = []
 353        nop90_indicator = 0
 354        nop6690_indicator = False
 355        fnop_indicator = False
 356        mov_indicator = False
 357        avg_cap = 0.0       # In BITS
 358        min_cap = 0         # In BITS
 359        max_cap = 0         # In BITS
 360        
 361        # File is opened because of more than 3 bytes long NOP class and
 362        # is closed at the end of Selector.
 363        try:
 364            fd = open(fexe, "rb")
 365        except IOError:
 366            print(f"ERROR! Can not open cover-file for analysis: ", file=sys.stderr)
 367            sys.exit(102)
 368        
 369        for my_instr in all_my_instrs:
 370            
 371            instr = my_instr.instruction
 372            op_code = my_instr.instruction.op_code()
 373
 374            ## NOT IMPLEMENTED YET.
 375            if method == "mov":
 376                
 377                # Two MOVs in a row.
 378                if instr.mnemonic == Mnemonic.MOV:
 379                    if mov_indicator:
 380                        # Got two MOVs in a row.
 381                        # Need to check their independency.
 382                        prev_mov = all_my_instrs[my_instr.ioffset - 1]
 383                        
 384                        if cls.__check_independency(prev_mov.instruction,
 385                                                    instr,
 386                                                    analyzer.bitness):
 387                            
 388                            if prev_mov.eq_class is not None and \
 389                                (
 390                                    prev_mov.eq_class.class_name == "MOV" or \
 391                                    prev_mov.eq_class.class_name == \
 392                                        "Swap base-index registers 32-bit"
 393                                ):
 394                                # Previous MOV has already got eq_class
 395                                # and also was selected.
 396                                prev_mov.set_mov_scheduling_flag = True
 397                                
 398                            else:
 399                                # Previous MOV has not any eq_class and
 400                                # was not selected yet.
 401                                cls.__set_eq_class(prev_mov, "MOV Scheduling")
 402                                selected_my_instrs.append(prev_mov)
 403                            
 404                            # Current MOV will be checked similarly in
 405                            # MOV equivalent class.
 406                            cls.__set_eq_class(my_instr, "MOV Scheduling")
 407                            selected_my_instrs.append(my_instr)
 408                            
 409                            # Reset indicator only if MOV instructions
 410                            # are independent and can be used, otherwise
 411                            # if next instruction will be MOV,
 412                            # dependency check will be performed again.
 413                            mov_indicator = False
 414                            
 415                            # Compute capacities.
 416                            avg_cap += my_instr.eq_class.avg_cap
 417                            min_cap += my_instr.eq_class.min_cap
 418                            max_cap += my_instr.eq_class.max_cap
 419                            
 420                    else:
 421                        # Set indicator.
 422                        mov_indicator = True
 423                else:
 424                    # Reset indicator.
 425                    mov_indicator = False
 426
 427
 428            if method == "nops" or \
 429                method == "ext-sub-nops":
 430                
 431                # 1 byte NOPs 0x90.
 432                if op_code.is_nop and instr.len == 1:
 433                    
 434                    if nop90_indicator > 0:
 435                        # 2nd or 3rd NOP 0x90 detected.
 436                        
 437                        if nop90_indicator == 1:
 438                            # 2nd NOP 0x90 detected.
 439                            nop90_indicator = 2
 440                            
 441                            prev_nop = all_my_instrs[my_instr.ioffset - 1]
 442                            cls.__set_eq_class(prev_nop, "2 Bytes Long NOP")
 443                            selected_my_instrs.append(prev_nop)
 444                            
 445                            cls.__set_eq_class(my_instr, "2 Bytes Long NOP")
 446                            selected_my_instrs.append(my_instr)
 447                        
 448                            # Compute capacities.
 449                            avg_cap += my_instr.eq_class.avg_cap
 450                            min_cap += my_instr.eq_class.min_cap
 451                            max_cap += my_instr.eq_class.max_cap
 452                        
 453                        elif nop90_indicator == 2:
 454                            # 3rd NOP 0x90 detected.
 455                            
 456                            # Reset indicator.
 457                            nop90_indicator = 0
 458                            
 459                            # Computation must also be returned.
 460                            avg_cap -= selected_my_instrs[-1].eq_class.avg_cap
 461                            min_cap -= selected_my_instrs[-1].eq_class.min_cap
 462                            max_cap -= selected_my_instrs[-1].eq_class.max_cap
 463                            
 464                            cls.__set_eq_class(
 465                                selected_my_instrs[-2], "3 Bytes Long NOP")
 466                            cls.__set_eq_class(
 467                                selected_my_instrs[-1], "3 Bytes Long NOP")
 468                            cls.__set_eq_class(my_instr, "3 Bytes Long NOP")
 469                            selected_my_instrs.append(my_instr)
 470                            
 471                            # Fix computation.
 472                            avg_cap += my_instr.eq_class.avg_cap
 473                            min_cap += my_instr.eq_class.min_cap
 474                            max_cap += my_instr.eq_class.max_cap
 475                        
 476                    else:
 477                        # 1st NOP 0x90 detected.
 478                        nop90_indicator = 1
 479                        
 480                        if nop6690_indicator:
 481                            # Detected sequence: 0x6690; 0x90.
 482                            
 483                            # Reset indicators.
 484                            nop90_indicator = 0
 485                            nop6690_indicator = False
 486
 487                            # Computation must also be returned.
 488                            avg_cap -= selected_my_instrs[-1].eq_class.avg_cap
 489                            min_cap -= selected_my_instrs[-1].eq_class.min_cap
 490                            max_cap -= selected_my_instrs[-1].eq_class.max_cap
 491                            
 492                            cls.__set_eq_class(

 494                            cls.__set_eq_class(my_instr, "3 Bytes Long NOP")
 495                            selected_my_instrs.append(my_instr)

 497                            # Fix computation.

 519                            # Fix computation.
 520                            avg_cap += my_instr.eq_class.avg_cap

 522                            max_cap += my_instr.eq_class.max_cap
 523                    
 524                # Multi-byte NOP.
 525                if op_code.is_nop:
 526                    
 527                    if instr.len == 2:
 528                        # 2 bytes NOP 0x6690 detected.
 529                        nop6690_indicator = True
 530                        
 531                        if nop90_indicator == 1:
 532                            # Detected sequence: 0x90; 0x6690
 533                            
 534                            nop6690_indicator = False
 535

 542                            # Compute capacities.
 543                            avg_cap += my_instr.eq_class.avg_cap
 544                            min_cap += my_instr.eq_class.min_cap
 545                            max_cap += my_instr.eq_class.max_cap
 546                            
 547                        else:                            
 548                            # Only one NOP 0x6690 detected.
 549                            cls.__set_eq_class(my_instr, "2 Bytes Long NOP")
 550                            selected_my_instrs.append(my_instr)
 551                            
 552                            # Compute capacities.
 553                            avg_cap += my_instr.eq_class.avg_cap
 554                            min_cap += my_instr.eq_class.min_cap
 555                            max_cap += my_instr.eq_class.max_cap
 556                            
 557                        # Reset indicators for a case they are set.
 558                        nop90_indicator = 0
 559                        fnop_indicator = False
 560
 561                    elif instr.len == 3:
 562                        # 3 bytes NOP 0x0f1f00.
 563                        
 564                        # Reset indicators for a case they are set.
 565                        nop90_indicator = 0
 566                        nop6690_indicator = False
 567                        fnop_indicator = False
 568                        
 569                        cls.__set_eq_class(my_instr, "3 Bytes Long NOP")
 570                        selected_my_instrs.append(my_instr)
 571                        
 572                        # Compute capacities.
 573                        avg_cap += my_instr.eq_class.avg_cap
 574                        min_cap += my_instr.eq_class.min_cap
 575                        max_cap += my_instr.eq_class.max_cap
 576                    
 577                    elif instr.len > 3:
 578                        # More than 3 bytes long NOP.
 579                        
 580                        # Reset indicators for a case they are set.
 581                        nop90_indicator = 0
 582                        nop6690_indicator = False
 583                        fnop_indicator = False
 584                        
 585                        cls.__set_eq_class(my_instr, ">3 Bytes Long NOP")

 627                        
 628                        # Compute capacities.
 629                        avg_cap += my_instr.eq_class.avg_cap
 630                        min_cap += my_instr.eq_class.min_cap
 631                        max_cap += my_instr.eq_class.max_cap
 632                        
 633                    # Reset indicators for a case they are set.
 634                    nop90_indicator = 0
 635                    nop6690_indicator = False
 636                        
 637                else:
 638                    # Reset indicators when non-NOP instruction occurs.
 639                    nop90_indicator = 0

 641                    fnop_indicator = False
 642                        
 643

 102    def __liveness_flags_detection(all_instrs: list,
 103                                   my_instr: MyInstruction,
 104                                   rflags_to_check: int,
 105                                   force_flag: bool) -> bool:
 106        """
 107        Liveness detection of modified flags by replacement instruction.
 108        Check given flags if they are live until any instruction which
 109        modifies it come. Execution flow is respected while checking.
 110        Flag is live if instruction which reads it come earlier than
 111        instruction which modofies it. Basic checking is done until
 112        any jump is present. If 'force' flag was given by user,
 113        unconditional jumps are traced and this can achieve bigger
 114        capacity. The end of any function stops checking with positive
 115        result.
 116        """
 117        if rflags_to_check == RflagsBits.NONE:
 118            return True
 119        
 120        all_rflags = (RflagsBits.OF, RflagsBits.SF, RflagsBits.ZF,
 121                      RflagsBits.AF, RflagsBits.CF, RflagsBits.PF,
 122                      RflagsBits.DF, RflagsBits.IF, RflagsBits.AC,
 123                      RflagsBits.UIF)
 124        
 125        # Transformation of given flags all in one integer to the list
 126        # of proper flags values.
 127        check_rflags = [rflag
 128                        for rflag in all_rflags
 129                            if (rflag & rflags_to_check) != 0]
 130        
 131        # List for marking whether the flag has already been checked
 132        # (modified by another instruction).
 133        safe_rflags = [False] * len(check_rflags)
 134
 135        # In 'my_instr' is always present next instruction from
 136        # previous one.
 137        my_instr = all_instrs[my_instr.ioffset + 1]
 138        
 139        while True:
 140            
 141            rflags_read = my_instr.instruction.rflags_read
 142            rflags_modified = my_instr.instruction.rflags_modified
 143
 144            for idx, rflag in enumerate(check_rflags):
 145                # Liveness detection of checked flags.
 146                
 147                if (rflag & rflags_read) != 0:
 148                    # Instruction tests checked flag.
 149                    return False
 150                
 151                if (rflag & rflags_modified) != 0:
 152                    # Instruction modifies checked flag.
 153                    safe_rflags[idx] = True
 154                    
 155                    # End immediately when all flags are safe.
 156                    if False not in safe_rflags:
 157                        return True
 158            
 159            # Only because of clearer conditions.
 160            instr = my_instr.instruction
 161            
 162            if instr.flow_control == FlowControl.NEXT:
 163                # Execution flow follow next instruction.
 164                my_instr = all_instrs[my_instr.ioffset + 1]
 165            
 166            elif instr.flow_control == FlowControl.RETURN:
 167                # The end of a function.
 168                break
 169            
 170            # CALL|JMP near/far
 171            # ONLY if user specify by argument (very time-consuming).
 172            elif force_flag and \
 173                (
 174                    instr.flow_control == FlowControl.CALL or \
 175                    instr.flow_control == FlowControl.UNCONDITIONAL_BRANCH
 176                ):
 177                   
 178                if instr.op_count == 1:
 179                    if instr.op0_kind == OpKind.NEAR_BRANCH16 or \
 180                        instr.op0_kind == OpKind.NEAR_BRANCH32 or \
 181                        instr.op0_kind == OpKind.NEAR_BRANCH64:
 182
 183                        get_target = instr.near_branch_target
 184                    
 185                    elif instr.op0_kind == OpKind.FAR_BRANCH16 or \
 186                        instr.op0_kind == OpKind.FAR_BRANCH32:
 187                        
 188                        get_target = instr.far_branch_selector
 189                        
 190                    else:
 191                        return False
 192                
 193                else:
 194                    return False
 195                    
 196                found_instr = [my_instr 
 197                                for my_instr in all_instrs 
 198                                    if my_instr.instruction.ip == get_target]
 199                
 200                if found_instr:
 201                    my_instr = found_instr[0]
 202                else:
 203                    return False
 204            
 205            else:   
 206                return False
 207
 208        return True

 238                         reg: Register_) -> bool:
 239        """
 240        Decide if memory operand of given instruction uses register `reg`.
 241        """
 242        if instr.memory_base != Register.NONE and \
 243            cls.__check_regs_dependency(bitness, instr.memory_base, reg):
 244            return True
 245        if instr.memory_index != Register.NONE and \
 246            cls.__check_regs_dependency(bitness, instr.memory_index, reg):
 247            return True
 248        
 249        return False
 250
 251    
 252    # ak do registra zapisujem, nesmiem uz z neho citat !!! ak sa nachadza v MEMORY, je to citanie !!! citat z neho mozem len v tej istej instruckii .
 253    # ak zapisujem do MEM, z tohto miesta uz nesmiem citat ani tam zapisat.
 254    @classmethod
 255    def __check_independency(cls,
 256                             prev_mov: Instruction,
 257                             curr_mov: Instruction,
 258                             bitness: int) -> bool:
 259        """
 260        Two registers are dependent on each other if their order is
 261        irreplaceable.
 262        
 263        If write to any register is performed, it can not be used
 264        anymore (bith sides). If register occurs on left side as
 265        memory operand, it's reading from it and can be used next time
 266        (MOV rax, rax; MOV rax, [rax]).
 267        
 268        From specific register or memory location can be read more
 269        times. If it's writing to any memory location, this location
 270        can not be used anymore.
 271        """
 272        
 273        # Can not be exactly two same MOV instructions.
 274        if prev_mov.eq_all_bits(curr_mov):
 275            return False
 276        # In case when they are not equal in bits, because of they
 277        # differ only in Direction Bit as MOV can -- in that case they
 278        # actually are equal and can not be ordered lexicographically.
 279        elif f"{prev_mov:ix}" == f"{curr_mov:ix}":
 280            return False
 281        
 282        # If some mov write to the REG, this REG can not be used anymore
 283        # (except within same instruction).
 284        if prev_mov.op0_kind == OpKind.REGISTER:
 285            
 286            if curr_mov.op0_kind == OpKind.REGISTER and \
 287                cls.__check_regs_dependency(bitness, curr_mov.op0_register, prev_mov.op0_register):
 288                return False
 289            if curr_mov.op1_kind == OpKind.REGISTER and \
 290                cls.__check_regs_dependency(bitness, curr_mov.op1_register, prev_mov.op0_register):
 291                return False
 292            if (
 293                    curr_mov.op0_kind == OpKind.MEMORY or \
 294                    curr_mov.op1_kind == OpKind.MEMORY
 295                ) and \
 296                cls.__mem_op_use_reg(bitness, curr_mov, prev_mov.op0_register):
 297                return False
 298        
 299        if curr_mov.op0_kind == OpKind.REGISTER:
 300            
 301            if prev_mov.op1_kind == OpKind.REGISTER and \
 302                cls.__check_regs_dependency(bitness, prev_mov.op1_register, curr_mov.op0_register):
 303                    return False
 304            if (
 305                    prev_mov.op0_kind == OpKind.MEMORY or \
 306                    prev_mov.op1_kind == OpKind.MEMORY
 307                ) and \
 308                cls.__mem_op_use_reg(bitness, prev_mov, curr_mov.op0_register):
 309                return False
 310            
 311        # Same MEMORY locations on the 'cross'.
 312        if prev_mov.op0_kind == OpKind.MEMORY:
 313            if prev_mov.op0_kind == curr_mov.op1_kind and \
 314                cls.__check_regs_dependency(bitness, prev_mov.memory_base, curr_mov.memory_base) and \
 315                cls.__check_regs_dependency(bitness, prev_mov.memory_index, curr_mov.memory_index) and \
 316                prev_mov.memory_index_scale == curr_mov.memory_index_scale and \
 317                prev_mov.memory_displacement == curr_mov.memory_displacement:
 318                return False
 319        
 320        # Same MEMORY locations on the 'cross'.
 321        if prev_mov.op1_kind == OpKind.MEMORY:
 322            if prev_mov.op1_kind == curr_mov.op0_kind and \
 323                cls.__check_regs_dependency(bitness, prev_mov.memory_base, curr_mov.memory_base) and \
 324                cls.__check_regs_dependency(bitness, prev_mov.memory_index, curr_mov.memory_index) and \
 325                prev_mov.memory_index_scale == curr_mov.memory_index_scale and \
 326                prev_mov.memory_displacement == curr_mov.memory_displacement:
 327                return False
 328            
 329        return True

 375            ## NOT IMPLEMENTED YET.

 446                            cls.__set_eq_class(my_instr, "2 Bytes Long NOP")

 450                            avg_cap += my_instr.eq_class.avg_cap
 451                            min_cap += my_instr.eq_class.min_cap
 452                            max_cap += my_instr.eq_class.max_cap
 453                        
 454                        elif nop90_indicator == 2:
 455                            # 3rd NOP 0x90 detected.
 456                            
 457                            # Reset indicator.
 458                            nop90_indicator = 0
 459                            
 460                            # Computation must also be returned.
 461                            avg_cap -= selected_my_instrs[-1].eq_class.avg_cap
 462                            min_cap -= selected_my_instrs[-1].eq_class.min_cap
 463                            max_cap -= selected_my_instrs[-1].eq_class.max_cap
 464                            
 465                            cls.__set_eq_class(
 466                                selected_my_instrs[-2], "3 Bytes Long NOP")
 467                            cls.__set_eq_class(
 468                                selected_my_instrs[-1], "3 Bytes Long NOP")
 469                            cls.__set_eq_class(my_instr, "3 Bytes Long NOP")
 470                            selected_my_instrs.append(my_instr)

 472                            # Fix computation.
 473                            avg_cap += my_instr.eq_class.avg_cap
 474                            min_cap += my_instr.eq_class.min_cap
 475                            max_cap += my_instr.eq_class.max_cap
 476                        
 477                    else:
 478                        # 1st NOP 0x90 detected.
 479                        nop90_indicator = 1
 480                        
 481                        if nop6690_indicator:
 482                            # Detected sequence: 0x6690; 0x90.
 483                            
 484                            # Reset indicators.
 485                            nop90_indicator = 0
 486                            nop6690_indicator = False
 487
 488                            # Computation must also be returned.
 489                            avg_cap -= selected_my_instrs[-1].eq_class.avg_cap
 490                            min_cap -= selected_my_instrs[-1].eq_class.min_cap
 491                            max_cap -= selected_my_instrs[-1].eq_class.max_cap

 493                            cls.__set_eq_class(
 494                                selected_my_instrs[-1], "3 Bytes Long NOP")
 495                            cls.__set_eq_class(my_instr, "3 Bytes Long NOP")
 496                            selected_my_instrs.append(my_instr)
 497                            
 498                            # Fix computation.
 499                            avg_cap += my_instr.eq_class.avg_cap
 500                            min_cap += my_instr.eq_class.min_cap
 501                            max_cap += my_instr.eq_class.max_cap
 502                        
 503                        elif fnop_indicator:
 504                            # Detected sequence: 0xd9d0; 0x90.
 505                            
 506                            # Reset indicators.
 507                            nop90_indicator = 0
 508                            fnop_indicator = False
 509                            
 510                            # Computation must also be returned.
 511                            avg_cap -= selected_my_instrs[-1].eq_class.avg_cap
 512                            min_cap -= selected_my_instrs[-1].eq_class.min_cap
 513                            max_cap -= selected_my_instrs[-1].eq_class.max_cap

 515                            cls.__set_eq_class(

 517                            cls.__set_eq_class(my_instr, "3 Bytes Long NOP")

 520                            # Fix computation.

 532                        if nop90_indicator == 1:
 533                            # Detected sequence: 0x90; 0x6690
 534                            
 535                            nop6690_indicator = False
 536

 546                            max_cap += my_instr.eq_class.max_cap

 549                            # Only one NOP 0x6690 detected.

 552                            
 553                            # Compute capacities.
 554                            avg_cap += my_instr.eq_class.avg_cap
 555                            min_cap += my_instr.eq_class.min_cap
 556                            max_cap += my_instr.eq_class.max_cap
 557                            
 558                        # Reset indicators for a case they are set.
 559                        nop90_indicator = 0
 560                        fnop_indicator = False
 561
 562                    elif instr.len == 3:
 563                        # 3 bytes NOP 0x0f1f00.
 564                        
 565                        # Reset indicators for a case they are set.
 566                        nop90_indicator = 0
 567                        nop6690_indicator = False
 568                        fnop_indicator = False
 569                        
 570                        cls.__set_eq_class(my_instr, "3 Bytes Long NOP")
 571                        selected_my_instrs.append(my_instr)
 572                        
 573                        # Compute capacities.
 574                        avg_cap += my_instr.eq_class.avg_cap
 575                        min_cap += my_instr.eq_class.min_cap
 576                        max_cap += my_instr.eq_class.max_cap
 577                    
 578                    elif instr.len > 3:
 579                        # More than 3 bytes long NOP.
 580                        
 581                        # Reset indicators for a case they are set.

 583                        nop6690_indicator = False
 584                        fnop_indicator = False

 612                        prev_nop = all_my_instrs[my_instr.ioffset - 1]

 704                    else:
